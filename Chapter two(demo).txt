//preprocessor directives
/*preprocessor directives are instructions to the compiler that are executed before the actual compilation process, and they are identified by a # symbol at the beginning of the line.*/
//example

#define PI 3.14

    double radius = 5.0;
    double area = PI * radius * radius;

    return 0;
}

//relational operators 
/*Relational operators in C++ are used to compare two values and return a boolean value (true or false) based on the result of the comparison.*/
//example
#include <iostream>
using namespace std;
int main() {
    int a = 10;
    int b = 5;
    // greater than operator
    if (a > b) {
        cout << "a is greater than b" << endl;
    }
    // less than operator
    if (b < a) {
        cout << "b is less than a" << endl;
    }
    // greater than or equal to operator
    if (a >= b) {
        cout << "a is greater than or equal to b" << endl;
    }
    // less than or equal to operator
    if (b <= a) {
        cout << "b is less than or equal to a" << endl;
    }
    // equal to operator
    if (a == b) {
        cout << "a is equal to b" << endl;
    }
    // not equal to operator
    if (a != b) {
        cout << "a is not equal to b" << endl;
    }
    return 0;
}


//logical operators 
//are used to determine the logic between variables or values:
//example
#include <iostream>
using namespace std;
int main() {
   int x = 10;
   int y = 20;
   if (x == 10 && y == 20) {
      cout << "x is equal to 10 and y is equal to 20" << endl;
   }  
   if (x == 10 || y == 30) {
      cout << "Either x is equal to 10 or y is equal to 30" << endl;
   }  
   if (!(x == 5)) {
      cout << "x is not equal to 5" << endl;
   }
   return 0;
}


//bitwise operators 
//Bitwise operators in C++ are used to manipulate individual bits within a value.
//example
#include <iostream>
using namespace std;
int main() {
    int a = 5;     // 0101 in binary
    int b = 3;     // 0011 in binary
    int c = a & b; // 0001 in binary (result of bitwise AND)
    cout <<"c is="<<c<<endl;   // prints 1
    return 0;
}



//increment/decrement operators 
#include <iostream>
using namespace std;
int main() {
    int a = 5;
    int b = a++; // b = 5, a = 6
    int c = ++a; // c = 7, a = 7
    int d = 10;
    int e = --d; // e = 9, d = 9
    int f = d--; // f = 9, d = 8
    cout << b << " " << c << " " << e << " " << f;
    return 0;
}

//precedence  of operators
//are used to determine the order in which operations are executed in an expression.
//example
#include <iostream>
using namespace std;
int main() {
    int x = 5, y = 10, z = 15;
    int result1 = x + y * z; // Multiply y and z first, then add x
    int result2 = (x + y) * z; // Add x and y first, then multiply by z
    cout << "Result 1: " << result1 << endl; // Output: 155
    cout << "Result 2: " << result2 << endl; // Output: 225
    return 0;
}

//simple type convesion
//the process of converting a value of one data type to another data type.

//example

#include <iostream>
using namespace std;
int main() {
    // Implicit conversion
    int x = 5;
    float y = x;
    cout << "Implicit conversion: " << x << " (int) to " << y << " (float)" << endl;
    
    // Explicit conversion
    float a = 3.14;
    int b = (int) a;
    cout << "Explicit conversion: " << a << " (float) to " << b << " (int)" << endl;   
    return 0;
}
